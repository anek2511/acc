  <script>

    document.addEventListener("DOMContentLoaded", () => {

      const today = new Date().toISOString().split('T')[0];

      document.getElementById('inc-date').value = today;

      document.getElementById('exp-date').value = today;

      switchMenu('income-menu');

    });


    let toastTimeout;

    function showToast(msg, type = 'success') {

      const toast = document.getElementById('toast');

      document.getElementById('toast-message').innerText = msg;

      if(type === 'error') {

        toast.classList.replace('bg-emerald-600', 'bg-red-600');

      } else {

        toast.classList.replace('bg-red-600', 'bg-emerald-600');

      }

      toast.classList.remove('translate-x-full', 'opacity-0');

      clearTimeout(toastTimeout);

      toastTimeout = setTimeout(() => {

        toast.classList.add('translate-x-full', 'opacity-0');

      }, 2000);

    }


    function switchMenu(menuId) {

      document.querySelectorAll('.menu-content').forEach(el => el.classList.add('hidden'));

      document.getElementById(menuId).classList.remove('hidden');

      

      document.querySelectorAll('.nav-btn').forEach(el => el.classList.remove('bg-indigo-700', 'ring-2', 'ring-white'));

      if(menuId === 'income-menu') document.getElementById('btn-income').classList.add('bg-indigo-700', 'ring-2', 'ring-white');

      if(menuId === 'expense-menu') document.getElementById('btn-expense').classList.add('bg-indigo-700', 'ring-2', 'ring-white');

      if(menuId === 'report-menu') document.getElementById('btn-report').classList.add('bg-indigo-700', 'ring-2', 'ring-white');

    }


    // --- บันทึกรายรับ ---

    function handleIncomeSubmit(e) {

      e.preventDefault();

      const btn = document.getElementById('inc-submit-btn');

      btn.disabled = true; btn.innerText = "กำลังบันทึก...";

      

      const data = {

        date: document.getElementById('inc-date').value,

        name: document.getElementById('inc-name').value,

        amount: document.getElementById('inc-amount').value,

        note: document.getElementById('inc-note').value

      };

      

      google.script.run.withSuccessHandler(res => {

        btn.disabled = false; btn.innerText = "บันทึกรายรับ";

        if(res.success) {

          showToast('บันทึกข้อมูลเรียบร้อยแล้ว');

          document.getElementById('inc-name').value = '';

          document.getElementById('inc-amount').value = '';

          document.getElementById('inc-note').value = '';

          document.getElementById('inc-name').focus();

        } else {

          showToast('เกิดข้อผิดพลาด: ' + res.error, 'error');

        }

      }).saveIncome(data);

    }


    // --- บันทึกรายจ่าย ---

    function handleExpenseSubmit(e) {

      e.preventDefault();

      const btn = document.getElementById('exp-submit-btn');

      btn.disabled = true; btn.innerText = "กำลังบันทึก...";

      

      const data = {

        date: document.getElementById('exp-date').value,

        item: document.getElementById('exp-item').value,

        amount: document.getElementById('exp-amount').value,

        note: document.getElementById('exp-note').value

      };

      

      google.script.run.withSuccessHandler(res => {

        btn.disabled = false; btn.innerText = "บันทึกรายจ่าย";

        if(res.success) {

          showToast('บันทึกรายจ่ายเรียบร้อยแล้ว');

          document.getElementById('exp-item').value = '';

          document.getElementById('exp-amount').value = '';

          document.getElementById('exp-note').value = '';

          document.getElementById('exp-item').focus();

        } else {

          showToast('เกิดข้อผิดพลาด: ' + res.error, 'error');

        }

      }).saveExpense(data);

    }


    // --- โหลดตารางรายงานสรุป ---

    function loadReport() {

      const tbody = document.getElementById('report-table-body');

      tbody.innerHTML = `<tr><td colspan="4" class="px-4 py-8 text-center text-gray-500">กำลังคำนวณ...</td></tr>`;


      google.script.run.withSuccessHandler(data => {

        document.getElementById('total-income').innerText = data.totalIncome.toLocaleString('th-TH', {minimumFractionDigits: 2});

        document.getElementById('total-expense').innerText = data.totalExpense.toLocaleString('th-TH', {minimumFractionDigits: 2});

        document.getElementById('total-balance').innerText = data.totalBalance.toLocaleString('th-TH', {minimumFractionDigits: 2});

        

        tbody.innerHTML = '';

        const dates = Object.keys(data.dailyData).sort((a, b) => new Date(b) - new Date(a)); 

        

        if(dates.length === 0) {

          tbody.innerHTML = `<tr><td colspan="4" class="px-4 py-4 text-center text-gray-400">ยังไม่มีข้อมูล</td></tr>`;

          return;

        }
